[binaries]
c = 'clang'
ar = 'arm-none-eabi-ar'
strip = 'arm-none-eabi-strip'
objcopy = 'arm-none-eabi-objcopy'
ld = 'arm-none-eabi-ld'
size = 'arm-none-eabi-size'

[built-in options]
c_args = ['--target=arm-none-eabi', '-mcpu=cortex-m4', '-mthumb', '-mfpu=fpv4-sp-d16', '-mfloat-abi=hard']
# Some environments default clang's bare-metal linker to `ld.lld`. We prefer
# GNU binutils (`arm-none-eabi-ld.bfd`) and explicitly select it so Meson can
# configure even when `ld.lld` isn't installed.
c_link_args = ['--target=arm-none-eabi', '-fuse-ld=bfd']
